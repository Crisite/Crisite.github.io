use sql_test;

#–1.学生表 
#Student(s_id,s_name,s_brith,s_sex) –学生编号,学生姓名, 出生年月,学生性别
CREATE TABLE `Student` (
    `s_id` VARCHAR(20),
    s_name VARCHAR(20) NOT NULL DEFAULT '',
    s_brith VARCHAR(20) NOT NULL DEFAULT '',
    s_sex VARCHAR(10) NOT NULL DEFAULT '',
    PRIMARY KEY(s_id)
);

#–2.课程表 
#Course(c_id,c_name,t_id) – –课程编号, 课程名称, 教师编号 
CREATE TABLE `Course`(
    c_id varchar(20),
    c_name VARCHAR(20) not null DEFAULT '',
    t_id VARCHAR(20) NOT NULL,
    PRIMARY KEY(c_id)
);

/*
–3.教师表 
Teacher(t_id,t_name) –教师编号,教师姓名 
*/
CREATE TABLE Teacher(
    t_id VARCHAR(20),
    t_name VARCHAR(20) NOT NULL DEFAULT '',
    PRIMARY KEY(t_id)
);

/*
–4.成绩表 
Score(s_id,c_id,s_score) –学生编号,课程编号,分数
*/
Create table Score(
    s_id VARCHAR(20),
    c_id VARCHAR(20) not null default '',
    s_score INT(3),
    primary key(`s_id`,`c_id`)
);

Sql习题

1、查询编号”01”课程比”02”课程成绩高的学生的信息及课程分数

需求格式

select student.* ,
       a.s_score as score_01,
       b.s_score as score_02
from student , score as a, score as b
where student.s_id = a.s_id
  and Student.s_id = b.s_id
  and a.c_id = '01'
  and b.c_id = '02'
  and a.s_score > b.s_score;

2、 查询”01”课程比”02”课程成绩低的学生的信息及课程分数

略

3、查询平均成绩大于等于60分的同学的学生编号和学生姓名和平均成绩

select Student.s_id, s_name, round(avg(s_score), 2)
from student left join score on Score.s_id = Student.s_id
group by Score.s_id
having avg(Score.s_score) >= 60;

4、查询平均成绩小于60分的同学的学生编号和学生姓名和平均成绩 (包括有成绩的和无成绩的)

略

5、查询所有同学的学生编号、学生姓名、选课总数、所有课程的总成绩（包含没有选课的）

select Student.s_id, s_name, count(s_score), sum(s_score)
from Student left join score s on Student.s_id = s.s_id
group by Student.s_id

6、查询”李”姓老师的数量

select count(t_id) from teacher where t_name like '李%';

7、查询学过”张三”老师授课的同学的信息

select student.* from student
    left join score s on Student.s_id = s.s_id
    left join course c on s.c_id = c.c_id
    left join teacher t on c.t_id = t.t_id
where t.t_name = '张三';

8、查询没学过”张三”老师授课的同学的信息

select student.* from student where student.s_id not in (
    select student.s_id from student
                          left join score s on Student.s_id = s.s_id
                          left join course c on s.c_id = c.c_id
                          left join teacher t on c.t_id = t.t_id
    where t.t_name = '张三');

9、查询学过编号为”01”并且也学过编号为”02”的课程的同学的信息

# 可以使用嵌套查询，也可以进行两次连表查询

select student.* from student join Score S on Student.s_id = S.s_id
where S.c_id = '01'
  AND Student.s_id in (
      select Student.s_id from Student join Score S on Student.s_id = S.s_id
      where S.c_id = '02'
);

13、查询和”01”号的同学学习的课程完全相同的其他同学的信息

两种思路：

  1. 使用group_concat函数，将课程列表输出为字符串，比较字符串是否相等
  2. 比较课程数量是否相等，再判断同学的课程是否全在01号同学的学习课程列表中
SELECT student.* from student
    left join Score S on Student.s_id = S.s_id
group by Student.s_id
having
       Student.s_id != '01'
       AND group_concat(s.c_id order by s.c_id) = (
    SELECT group_concat(s.c_id order by s.c_id)
    from score s
        join student s2 on s.s_id = s2.s_id
    where s2.s_id = '01'
);

# 思路二走不通 一种错误尝试
